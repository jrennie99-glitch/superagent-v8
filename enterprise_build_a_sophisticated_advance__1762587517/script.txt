// Function to calculate scientific functions
function calculateScientificFunction(func, value, angleUnit) {
  switch (func) {
    case 'sin':
      if (angleUnit === ANGLE_UNITS.DEGREES) {
        return math.sin(math.deg2rad(value));
      } else {
        return math.sin(value);
      }
    case 'cos':
      if (angleUnit === ANGLE_UNITS.DEGREES) {
        return math.cos(math.deg2rad(value));
      } else {
        return math.cos(value);
      }
    case 'tan':
      if (angleUnit === ANGLE_UNITS.DEGREES) {
        return math.tan(math.deg2rad(value));
      } else {
        return math.tan(value);
      }
    case 'asin':
      if (angleUnit === ANGLE_UNITS.DEGREES) {
        return math.rad2deg(math.asin(value));
      } else {
        return math.asin(value);
      }
    case 'acos':
      if (angleUnit === ANGLE_UNITS.DEGREES) {
        return math.rad2deg(math.acos(value));
      } else {
        return math.acos(value);
      }
    case 'atan':
      if (angleUnit === ANGLE_UNITS.DEGREES) {
        return math.rad2deg(math.atan(value));
      } else {
        return math.atan(value);
      }
    case 'log':
      return math.log(value);
    case 'ln':
      return math.ln(value);
    case 'log10':
      return math.log10(value);
    case 'sqrt':
      return math.sqrt(value);
    case 'cbrt':
      return math.cbrt(value);
    case 'exp':
      return math.exp(value);
    default:
      return 0;
  }
}
