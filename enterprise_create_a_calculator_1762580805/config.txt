  <script>
    function saveConfig() {
      const themeColor = document.getElementById('themeColor').value;
      const displayPrecision = document.getElementById('displayPrecision').value;
      const scientificNotationThreshold = document.getElementById('scientificNotationThreshold').value;
      const errorMessageElement = document.getElementById('errorMessage');
      const successMessageElement = document.getElementById('successMessage');


      // Validation (Example - Adapt to your needs)
      if (isNaN(scientificNotationThreshold) || scientificNotationThreshold <= 0) {
          errorMessageElement.textContent = "Invalid Threshold: Please enter a positive number.";
          successMessageElement.textContent = ""; // Clear success message
          return;
      } else {
        errorMessageElement.textContent = ""; //Clear previous error
      }
      

      try {
        localStorage.setItem('calculatorThemeColor', themeColor);
        localStorage.setItem('calculatorDisplayPrecision', displayPrecision);
        localStorage.setItem('calculatorScientificNotationThreshold', scientificNotationThreshold);

        successMessageElement.textContent = "Configuration saved successfully!";

        // Optional: Redirect back to the calculator, or update CSS dynamically here if it is in the same window.
        // window.location.href = "calculator.html"; // Replace with the actual calculator URL
      } catch (error) {
        errorMessageElement.textContent = "Error saving configuration: " + error.message;
        successMessageElement.textContent = "";
      }
    }

    // Optional: Load initial configuration when the page loads.  (Adapt to how your config is consumed)
    window.onload = function() {
        const savedThemeColor = localStorage.getItem('calculatorThemeColor');
        const savedDisplayPrecision = localStorage.getItem('calculatorDisplayPrecision');
        const savedScientificNotationThreshold = localStorage.getItem('calculatorScientificNotationThreshold');

        if (savedThemeColor) {
            document.getElementById('themeColor').value = savedThemeColor;
        }
        if (savedDisplayPrecision) {
            document.getElementById('displayPrecision').value = savedDisplayPrecision;
        }
        if (savedScientificNotationThreshold) {
          document.getElementById('scientificNotationThreshold').value = savedScientificNotationThreshold;
        }
    };
  </script>
