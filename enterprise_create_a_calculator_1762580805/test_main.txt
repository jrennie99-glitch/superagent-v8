    <script>
        /* script.js */
        let currentInput = document.getElementById('currentInput');
        let historyDisplay = document.getElementById('history');
        let calculationHistory = [];
        let isNewCalculation = true;

        document.addEventListener('keydown', function(event) {
            const key = event.key;

            if (!isNaN(parseInt(key)) || key === '.') {
                appendValue(key);
            } else if (key === '+' || key === '-' || key === '*' || key === '/') {
                // Replace * and / with proper symbols
                const operator = (key === '*') ? '×' : (key === '/') ? '÷' : key;
                appendValue(operator);
            } else if (key === 'Enter' || key === '=') {
                calculate();
            } else if (key === 'Escape') {
                clearDisplay();
            } else if (key === 'Backspace') {
                backspace();
            }
        });

        function appendValue(value) {
            if (isNewCalculation && !isNaN(parseFloat(value))) {
                currentInput.innerText = value;
                isNewCalculation = false;
            } else if (currentInput.innerText === '0' && value !== '.' && !isNaN(parseFloat(value))) {
                currentInput.innerText = value; // Replace initial zero
            } else {
                let currentText = currentInput.innerText;

                // Prevent multiple leading zeros if the current value is 0
                if (currentText === '0' && value === '0') {
                    return;
                }
                if (currentText.includes('.') && value === '.') {
                  return;
                }
                currentInput.innerText += value;
            }
        }

        function clearDisplay() {
            currentInput.innerText = '0';
            historyDisplay.innerText = '';
            isNewCalculation = true;
        }

        function backspace() {
            let currentText = currentInput.innerText;
            if (currentText.length > 1) {
                currentInput.innerText = currentText.slice(0, -1);
            } else {
                currentInput.innerText = '0';
                isNewCalculation = true;
            }
        }

        function toggleSign() {
            let currentValue = parseFloat(currentInput.innerText);
            if (!isNaN(currentValue)) {
                currentInput.innerText = (-currentValue).toString();
            }
        }

        function calculate() {
            try {
                let expression = currentInput.innerText;

                // Replace unicode symbols with JavaScript equivalents for evaluation
                expression = expression.replace(/×/g, '*').replace(/÷/g, '/').replace(/−/g, '-');

                // Sanitize input to prevent code injection (very basic)
                if (!isValidExpression(expression)) {
                    currentInput.innerText = 'Error';
                    historyDisplay.innerText = 'Invalid Expression';
                    return;
                }

                // Evaluate the expression
                let result = eval(expression);

                if (isNaN(result) || !isFinite(result)) {
                    currentInput.innerText = 'Error';
                    historyDisplay.innerText = 'Calculation Error';
                    return;
                }

                // Limit to a reasonable number of decimal places
                result = parseFloat(result.toFixed(10));

                // Update history
                const calculationString = `${currentInput.innerText} = ${result}`;
                calculationHistory.push(calculationString);
                localStorage.setItem('calculationHistory', JSON.stringify(calculationHistory));

                historyDisplay.innerText = expression;
                currentInput.innerText = result.toString();
                isNewCalculation = true;

            } catch (error) {
                console.error("Calculation error:", error);
                currentInput.innerText = 'Error';
                historyDisplay.innerText = 'Error';
            }
        }

        function isValidExpression(expression) {
            // Very basic check to prevent most common injection attempts
            // TODO: Implement a more robust validation
            return /^[0-9+\-*/().% ]+$/.test(expression);
        }

        // Load history from localStorage on page load
        window.onload = function() {
            const storedHistory = localStorage.getItem('calculationHistory');
            if (storedHistory) {
                calculationHistory = JSON.parse(storedHistory);
            }
        };

    </script>
